	GER_relocate_industry_civ = {
		icon = generic_construction
		# highlight_states = {
		# 	state = 19
		# }
		available = {
			NOT = {
				any_state = {
					OR = { is_core_of = FRF is_core_of = BEL is_core_of = HOL }

				}

			GER = { has_war_with = ENG }
		}

				}
				remove_building = {type = industrial_complex level = 1}

					random_owned_controlled_state = {
						limit = { is_core_of = GER }

   				 			type = industrial_complex

  				 			instant_build = yes
				 		}
					}
				}
				hidden_effect = {
					damage_building = { type = industrial_complex damage = 0.2 }
				}
			}
			hidden_effect = {
				if = {
					limit = {
						any_state = {
